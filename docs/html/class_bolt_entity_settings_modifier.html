Detailed Description

Modifier for bolt entity settings before it's attached

Example: Attaching an entity with custom settings

1 if (BoltNetwork.isServer) {
2  BoltEntity entity = gameObject.AddComponent&ltBoltEntity&gt();
3 
4  using (var mod = entity.ModifySettings()) {
5  mod.persistThroughSceneLoads = true;
6  mod.allowInstantiateOnClient = false;
7  mod.clientPredicted = false;
8  mod.prefabId = prefabId;
9  mod.updateRate = 1;
10  mod.sceneId = Bolt.UniqueId.None;
11  mod.serializerId = state;
12  }
13 
14  BoltNetwork.Attach(entity.gameObject);
15 }
